            <div className="text-6xl font-bold tabular-nums" aria-live="polite">
              {formatTime(time)}
            </div>
            <div className="flex gap-4">
              <Button onClick={isRunning ? handlePause : handleStart}>
                {isRunning ? 'Pause' : 'Start'}
              </Button>
              <Button onClick={handleReset}>Reset</Button>
            </div>
          </CardContent>
