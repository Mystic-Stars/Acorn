## ¶自动积分续期脚本

<ArticleCodeBlock code={"import requests\nimport json\nimport time\nimport datetime\n\ninstances = [\n # 在这里放置 [\"<X-API-KEY>\", \"<产品ID>\"]，可放置多项\n]\nurl_getinfo = \"https://api.v2.rainyun.com/product/rgs/{id}/\"\nurl_renew = \"https://api.v2.rainyun.com/product/point_renew\"\n\n# 一次续费的天数\nduration_day = 7\n \ndef GetRemainingDays(unix_timestamp):\n # 获取当前时间的UNIX时间戳\n current_timestamp = datetime.datetime.now().timestamp()\n # 将UNIX时间戳转换为datetime对象\n target_date = datetime.datetime.fromtimestamp(unix_timestamp)\n # 将当前时间的UNIX时间戳转换为datetime对象\n current_date = datetime.datetime.fromtimestamp(current_timestamp)\n # 计算两个日期之间的差异\n difference = current_date - target_date\n # 获取差异的天数部分\n days_difference = difference.days\n return -days_difference\n\nprint(\"Rainyun-PointRenew-V1 script, by CodeZhangBorui\\n[Time] \", end='')\nprint(time.ctime())\n\nfor instance in instances:\n key = instance[0]\n pid = instance[1] \n print(\"# 处理实例: API-KEY=\" + key[:10] + \"*\"*22 + \",产品ID=\" + pid)\n headers = {\n 'x-api-key': key,\n 'User-Agent': 'Rainyun-AutoRenew/1.0 (https://codezhangborui.eu.org/2023/06/rainyun-auto-python-scripts/)'\n }\n response = requests.request(\"GET\", url_getinfo.replace('{id}',pid), headers=headers, data={})\n result = json.loads(response.text)\n try:\n timestamp = result['data']['Data']['ExpDate']\n except:\n print(\"!
